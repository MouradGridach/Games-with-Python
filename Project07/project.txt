class ImageInfo:
    def __init__(self, center, size, radius = 0, lifespan = None, animated = False):
        self.center = center
        self.size = size
        self.radius = radius
        if lifespan:
            self.lifespan = lifespan
        else:
            self.lifespan = float('inf')
        self.animated = animated

    def get_center(self):
        return self.center

    def get_size(self):
        return self.size

    def get_radius(self):
        return self.radius

    def get_lifespan(self):
        return self.lifespan

    def get_animated(self):
        return self.animated

# helper functions to handle transformations
def angle_to_vector(ang):
    return [math.cos(ang), math.sin(ang)]

def dist(p,q):
    return math.sqrt((p[0] - q[0]) ** 2+(p[1] - q[1]) ** 2)


# Ship class
class Ship:
    def __init__(self, pos, vel, angle, image, info):
        self.pos = [pos[0],pos[1]]
        self.vel = [vel[0],vel[1]]
        self.thrust = False
        self.angle = angle
        self.angle_vel = 0
        self.image = image
        self.image_center = info.get_center()
        self.image_size = info.get_size()
        self.radius = info.get_radius()
        
    def draw(self,canvas):
        canvas.draw_image(self.image,self.image_center,self.image_size,self.pos,self.image_size,self.angle)
        if self.thrust:
            self.image_center = [135, 45]
        else:
            self.image_center = [45, 45]
    
    def shoot(self):
        global a_missile
        forward = angle_to_vector(self.angle)
        mis_pos = (forward[0] * self.radius, forward[1] * self.radius)
        p0 = self.pos[0] + mis_pos[0]
        p1 = self.pos[1] + mis_pos[1]
        
        a_missile.vel[0] = self.vel[0] + 12 * forward[0]
        a_missile.vel[1] = self.vel[1] + 12 * forward[1]
        a_missile = Sprite([p0, p1], a_missile.vel, 0, 0, missile_image, missile_info, missile_sound)        
        
    
    def increment(self):
        self.angle_vel += 0.09
    
    def decrement(self):
        self.angle_vel -= 0.09
        
    def set_thrusters(self, val):
        self.thrust = val
        if val:
            ship_thrust_sound.play()
        else:
            ship_thrust_sound.rewind()
    
    def update(self):
        global a_missile
        self.pos[0] += self.vel[0]
        self.pos[1] += self.vel[1]
        self.angle += self.angle_vel
        forward = angle_to_vector(self.angle)
        
        if self.pos[1] < 5:
            self.pos[1] = (self.pos[1] - 5) % HEIGHT
        elif self.pos[1] > 600:
            self.pos[1] = (self.pos[1] + 5) % HEIGHT
        if self.pos[0] > 800:
            self.pos[0] = (self.pos[0] + 5) % WIDTH
        elif self.pos[0] < 5:
            self.pos[0] = (self.pos[0] - 5) % WIDTH
        
        c = 0.07
        # Adding friction
        self.vel[0] *= (1 - c)
        self.vel[1] *= (1 - c)
        
        if self.thrust:
            self.vel[0] += forward[0]
            self.vel[1] += forward[1]
        
        if a_missile.pos[1] < 5:
            a_missile.pos[1] = (a_missile.pos[1] - 5) % HEIGHT
        elif a_missile.pos[1] > 600:
            a_missile.pos[1] = (a_missile.pos[1] + 5) % HEIGHT
        if a_missile.pos[0] > 800:
            a_missile.pos[0] = (a_missile.pos[0] + 5) % WIDTH
        elif a_missile.pos[0] < 5:
            a_missile.pos[0] = (a_missile.pos[0] - 5) % WIDTH
        
# Sprite class
class Sprite:
    def __init__(self, pos, vel, ang, ang_vel, image, info, sound = None):
        self.pos = [pos[0],pos[1]]
        self.vel = [vel[0],vel[1]]
        self.angle = ang
        self.angle_vel = ang_vel
        self.image = image
        self.image_center = info.get_center()
        self.image_size = info.get_size()
        self.radius = info.get_radius()
        self.lifespan = info.get_lifespan()
        self.animated = info.get_animated()
        self.age = 0
        if sound:
            sound.rewind()
            sound.play()
   
    def draw(self, canvas):
        canvas.draw_image(self.image, self.image_center, self.image_size, self.pos, self.image_size, self.angle)
    
    def update(self):
        self.angle += self.angle_vel
        self.pos[0] += self.vel[0]
        self.pos[1] += self.vel[1]


def keydown(key):
    if key == simplegui.KEY_MAP["left"]:
        my_ship.decrement()
    elif key == simplegui.KEY_MAP["right"]:
        my_ship.increment()
    elif key == simplegui.KEY_MAP["up"]:
        my_ship.set_thrusters(True)
    elif key == simplegui.KEY_MAP["space"]:
        my_ship.shoot()


def keyup(key):
    if key == simplegui.KEY_MAP["left"]:
        my_ship.angle_vel = 0
    elif key == simplegui.KEY_MAP["right"]:
        my_ship.angle_vel = 0
    elif key == simplegui.KEY_MAP["up"]:
        my_ship.set_thrusters(False)
        #my_ship.vel[0] = 0
    
def draw(canvas):
    global time, lives, score
    

    # animiate background
    time += 1
    center = debris_info.get_center()
    size = debris_info.get_size()
    wtime = (time / 8) % center[0]
    canvas.draw_image(nebula_image, nebula_info.get_center(), nebula_info.get_size(), [WIDTH / 2, HEIGHT / 2], [WIDTH, HEIGHT])
    canvas.draw_image(debris_image, [center[0] - wtime, center[1]], [size[0] - 2 * wtime, size[1]], 
                                [WIDTH / 2 + 1.25 * wtime, HEIGHT / 2], [WIDTH - 2.5 * wtime, HEIGHT])
    canvas.draw_image(debris_image, [size[0] - wtime, center[1]], [2 * wtime, size[1]], 
                                [1.25 * wtime, HEIGHT / 2], [2.5 * wtime, HEIGHT])
    
    canvas.draw_text("Lives", (50, 40), 30, "White")
    canvas.draw_text(str(lives), (50, 70), 30, "White")
    canvas.draw_text("Score", (650, 40), 30, "White")
    canvas.draw_text(str(score), (650, 70), 30, "White")

    # draw ship and sprites
    my_ship.draw(canvas)
    a_rock.draw(canvas)
    a_missile.draw(canvas)
    
    # update ship and sprites
    my_ship.update()
    a_rock.update()
    a_missile.update()
            
# timer handler that spawns a rock    
def rock_spawner():
    global a_rock
    wid = random.randrange(15, WIDTH - 15)
    hei = random.randrange(15, HEIGHT - 15)
    rock_av = random.randrange(0, 10) / 100 
    a_rock = Sprite([wid, hei], [random.randrange(-5, 5), random.randrange(-5,5)], 0, rock_av, asteroid_image, asteroid_info)
    
    
# initialize frame
frame = simplegui.create_frame("Asteroids", WIDTH, HEIGHT)
frame.set_keydown_handler(keydown)
frame.set_keyup_handler(keyup)

# initialize ship and two sprites
my_ship = Ship([WIDTH / 2, HEIGHT / 2], [1.7, 1.3], 0, ship_image, ship_info)
a_rock = Sprite([WIDTH / 3, HEIGHT / 3], [1, 1], 0, 0.09, asteroid_image, asteroid_info)
a_missile = Sprite([2 * WIDTH / 3, 2 * HEIGHT / 3], [-5,5], 0, 0, missile_image, missile_info, missile_sound)

# register handlers
frame.set_draw_handler(draw)

timer = simplegui.create_timer(1000.0, rock_spawner)

# get things rolling
timer.start()
frame.start()
